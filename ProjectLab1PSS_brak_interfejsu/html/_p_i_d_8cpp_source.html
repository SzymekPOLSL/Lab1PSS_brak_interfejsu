28double PID::Symuluj(double s_e) {
29 double u = 0.0;
30 //Aktualizacja wartości błędu
31 this->s_error_PI.push_front(s_e);
32 this->s_error_PI.pop_back();
33 //Obliczenie sumy błędu potrzebnej w części całkującej
34 this->s_sumaI_PI += this->s_error_PI.at(0);
35 //Obliczenie części proporcjonalnej, całkującej oraz różniczkującej
36 double P = PI::liczP();
37 double I = liczI();
38 double D = liczD();
39 u = P + I + D;
40 //Anti-wnidup
41 if (u > this->s_Umax_PI) {
42 this->s_sumaI_PI = (u - P - D)/(this->s_ki_PI * this->s_h_PI);
43 u = s_Umax_PI;
44 }
45 else if (u < this->s_Umin_PI) {
46 this->s_sumaI_PI = (u - P - D)/(this->s_ki_PI * this->s_h_PI);
