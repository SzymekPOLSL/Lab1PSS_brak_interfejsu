97/*
98* Symulacja modelu ARX. Na wejściu podawana jest pojedyncza próbka wejścia,
99* zaś poprzez symulację otrzymywana jest próbka wyjścia. Wyjście zaszumiane jest białym szumem.
100*
101*/
102double ARX::Symuluj(double u){
103 //Usunięcie wartości wejścia z tyłu i dodanie nowego wejścia z przodu.
104 this->s_u.push_front(u);
105 this->s_u.pop_back();
106 //Obliczenie wartości wyjścia w i-tej próbce
107 double a = std::inner_product(this->s_paramB.begin(), this->s_paramB.end(), this->s_u.begin()+s_k, 0.0);
108 double b = std::inner_product(this->s_paramA.begin(), this->s_paramA.end(), this->s_y.begin(), 0.0);
109 double wynik = a - b;
110 //Losowanie wartości, odwzorowującej szum (rozkład normalny)
111 std::random_device rd;
112 std::mt19937 gen(rd());
113 std::normal_distribution<> dist(0, 1);
114 double szum = dist(gen) * s_varE;
115 //Usunięcie wartości wyjścia z tyłu i dodanie nowego wyjścia wraz z szumem z przodu.
116 this->s_y.push_front(wynik+szum);
117 this->s_y.pop_back();
118 return wynik+szum;
119}
