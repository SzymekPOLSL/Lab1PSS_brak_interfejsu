103/*
104* Symulacja modelu ARX. Na wejściu podawana jest pojedyncza próbka wejścia,
105* zaś poprzez symulację otrzymywana jest próbka wyjścia. Wyjście zaszumiane jest białym szumem.
106*
107*/
108double ARX::Symuluj(double u){
109 //Usunięcie wartości wejścia z tyłu i dodanie nowego wejścia z przodu.
110 this->s_u.push_front(u);
111 this->s_u.pop_back();
112 //Obliczenie wartości wyjścia w i-tej próbce
113 double a = std::inner_product(this->s_paramB.begin(), this->s_paramB.end(), this->s_u.begin()+s_k, 0.0);
114 double b = std::inner_product(this->s_paramA.begin(), this->s_paramA.end(), this->s_y.begin(), 0.0);
115 double wynik = a - b;
116 //Losowanie wartości, odwzorowującej szum (rozkład normalny)
117 std::random_device rd;
118 std::mt19937 gen(rd());
119 std::normal_distribution<> dist(0, 1);
120 double szum = dist(gen) * s_varE;
121 //Usunięcie wartości wyjścia z tyłu i dodanie nowego wyjścia wraz z szumem z przodu.
122 this->s_y.push_front(wynik+szum);
123 this->s_y.pop_back();
124 return wynik;
125}
