37double PD::Symuluj(double s_y){

39 //Aktualizacja wartości błędu
40 this->s_error.push_front(this->s_w - s_y);
41 this->s_error.pop_back();
42 //Obliczenie części proporcjonalnej oraz różniczkującej
43 double P = this->s_kp * this->s_error.at(0);
44 double D = this->s_kd * (this->s_error.at(0) - this->s_error.at(1))/this->s_h;
45 u = P + D;
46 //Sprawdzenie, czy sterowanie mieści się w zadanym zakresie.
47 if (u > this->s_Umax) {
48 u = s_Umax;

51 u = s_Umin;
52 }
